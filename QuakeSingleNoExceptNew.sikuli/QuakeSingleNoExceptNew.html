def PersistentClick(img, imgOffs):
    try:
        click(img)
        wait(1)
        if exists(img):
            click(img)
            wait(1)
        else:
            return
        if exists(img):
            click(imgOffs)
            wait(1)
        else:
            return
        if exists(img):
            hover(imgOffs)
            wait(1)
            mouseDown(Button.LEFT)
            wait(1)
            mouseUp()
            wait(1)
    except:
        return
