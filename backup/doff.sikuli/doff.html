def AdmGetBestThree(tod = False):
    ids = AdmGetAvailableShips(tod)
    max_ids = [-1,-1,-1]
    if ids is None:
        return max_ids
    if tod:
        ships = adm_ships
    else:
        ships = adm_fleet
    for div in [1,2,3]:
        max = 0
        for id in ids:
            if ships[id][div] > max:
                max = ships[id][div]
                max_ids[div-1] = id
        if max_ids[div-1] in ids:
            ids.remove(max_ids[div-1])
    return max_ids
